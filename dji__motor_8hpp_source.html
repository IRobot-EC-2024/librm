144 template <DjiMotorType motor_type>
145 DjiMotor<motor_type>::DjiMotor(hal::CanBase &can, u16 id, bool reversed)
146  : CanDeviceBase(can, DjiMotorProperties<motor_type>::kRxIdBase + id), id_(id) {
147  // 如果这个电机对象所在CAN总线的发送缓冲区还未创建，就创建一个
148  if (DjiMotorProperties<motor_type>::tx_buf_.find(&can) == DjiMotorProperties<motor_type>::tx_buf_.end()) {
149  DjiMotorProperties<motor_type>::tx_buf_.insert({&can, {0}});
150  }

159 template <DjiMotorType motor_type>
160 void DjiMotor<motor_type>::SetCurrent(i16 current) {
161  // 限幅到电机能接受的最大电流
162  current = modules::algorithm::utils::absConstrain(current, DjiMotorProperties<motor_type>::kCurrentBound);
163  // 处理反转
164  if (this->reversed_) {
165  current = static_cast<i16>(-current);
166  }
167  // 根据这个电机所属的can总线(this->can_)和电机ID(this->id_)找到对应的发送缓冲区，写入电流值
168  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at((this->id_ - 1) * 2) = (current >> 8) & 0xff;
169  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at((this->id_ - 1) * 2 + 1) = current & 0xff;
170  // 根据电机ID判断缓冲区前八位需要发送，还是后八位需要发送，把对应的标志位置1
171  if (1 <= this->id_ && this->id_ <= 4) {
172  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at(16) = 1;
173  } else if (5 <= this->id_ && this->id_ <= 8) {
174  DjiMotorProperties<motor_type>::tx_buf_[this->can_].at(17) = 1;
175  }
176 }

181 template <DjiMotorType motor_type>
182 void DjiMotor<motor_type>::SendCommand() {
183  for (auto &buf : DjiMotorProperties<motor_type>::tx_buf_) {
184  if (buf.second.at(16) == 1) {
185  buf.first->Write(DjiMotorProperties<motor_type>::kControlId[0], buf.second.data(), 8);
186  buf.second.at(16) = 0;
187  }
188  if (buf.second.at(17) == 1) {
189  buf.first->Write(DjiMotorProperties<motor_type>::kControlId[1], buf.second.data() + 8, 8);
190  buf.second.at(17) = 0;
191  }
192  }
193 }

210 template <DjiMotorType motor_type>
211 void DjiMotor<motor_type>::RxCallback(const hal::CanRxMsg *msg) {
212  this->encoder_ = (msg->data[0] << 8) | msg->data[1];
213  if (this->reversed_) {
214  this->encoder_ = kDjiMotorMaxEncoder - this->encoder_;
215  }
216  this->rpm_ = (msg->data[2] << 8) | msg->data[3];
217  this->current_ = (msg->data[4] << 8) | msg->data[5];
218  this->temperature_ = msg->data[6];
219 }
