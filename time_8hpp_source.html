64 inline void SleepUs(u32 us) {
65  u32 ticks = 0;
66  u32 t_old = 0;
67  u32 t_now = 0;
68  u32 t_cnt = 0;
69  u32 reload = 0;
70  reload = SysTick->LOAD;
71  ticks = us * HAL_RCC_GetHCLKFreq() / 1000000;
72  t_old = SysTick->VAL;
73  while (true) {
74  t_now = SysTick->VAL;
75  if (t_now != t_old) {
76  if (t_now < t_old) {
77  t_cnt += t_old - t_now;
78  } else {
79  t_cnt += reload - t_now + t_old;
80  }
81  t_old = t_now;
82  if (t_cnt >= ticks) {
83  break;
84  }
85  }
86  }
87 }
88 #endif
