64 inline void SleepUs(u32 us) {
65  u32 ticks = 0;
66  u32 t_old = 0;
67  u32 t_now = 0;
68  u32 t_cnt = 0;
69  u32 reload = 0;
70  reload = SysTick->LOAD;
71  ticks = us * HAL_RCC_GetHCLKFreq() / 1000000;
72  t_old = SysTick->VAL;
73  while (true) {
74  t_now = SysTick->VAL;
75  if (t_now != t_old) {
76  if (t_now < t_old) {
77  t_cnt += t_old - t_now;
78  } else {
79  t_cnt += reload - t_now + t_old;
80  }
81  t_old = t_now;
82  if (t_cnt >= ticks) {
83  break;
84  }
85  }
86  }
87 }

108 void Sleep(T duration) {
109 #if defined(IROBOTEC_PLATFORM_STM32)
110  if constexpr (std::is_same_v<T, std::chrono::hours>) {
111  SleepMs(duration.count() * 3600000);
112  } else if constexpr (std::is_same_v<T, std::chrono::minutes>) {
113  SleepMs(duration.count() * 60000);
114  } else if constexpr (std::is_same_v<T, std::chrono::seconds>) {
115  SleepMs(duration.count() * 1000);
116  } else if constexpr (std::is_same_v<T, std::chrono::milliseconds>) {
117  SleepMs(duration.count());
118  } else if constexpr (std::is_same_v<T, std::chrono::microseconds>) {
119  SleepUs(duration.count());
120  } else if constexpr (std::is_same_v<T, std::chrono::nanoseconds>) {
121  SleepUs(duration.count() / 1000);
122  }
123 #else
124  std::this_thread::sleep_for(duration);
125 #endif
126 }
